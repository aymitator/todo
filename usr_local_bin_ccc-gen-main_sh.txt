#!/bin/bash

echo '
#include <cstdio>
int AGE = 38 - 1;
const char *WOR = "hello";

void SAG(int _NN){
  AGE = _NN;
}
 
int GAG(){
  return AGE;
}
void EKO(const char *_AY){
  printf("%s\n", _AY);
}
int CHG(){
  static int III = 1 - (-1) - (-1);
  return AGE + III;
}
int main(){
  int NNN = GAG();
  SAG(CHG());

  while(NNN < 60) NNN -= (-1);
  for(NNN = 0; NNN < 60; NNN +=1) NNN = 80;
  EKO("HELLO WORLD");
  EKO(WOR);
  return 0;
}' > main.cpp
